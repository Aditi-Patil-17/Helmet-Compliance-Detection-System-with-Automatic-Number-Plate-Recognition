% --- STANDALONE EVALUATION: CONFUSION MATRIX ---
clear; clc;

% 1. Load the required files
if exist('augmented_helmet_data.mat', 'file') && exist('trainedHelmetModel_90.mat', 'file')
    load('augmented_helmet_data.mat'); % Contains training_features and training_labels
    load('trainedHelmetModel_90.mat');  % Contains mlModel
else
    error('Missing .mat files! Run the Training and Augmentation scripts first.');
end

% 2. Use the loaded features and labels for evaluation
X_eval = training_features; 
Y_eval = training_labels;   

% 3. Generate Predictions using the ensemble
[predicted_labels, ~] = predict(mlModel, X_eval); 

% 4. Visualize the Confusion Matrix for the IEEE Paper
figure('Color', 'w', 'Name', 'Model Performance Analysis');
cm = confusionchart(Y_eval, predicted_labels, ...
    'Title', 'TABLE III. CONFUSION MATRIX OF HYBRID CLASSIFIER', ...
    'ColumnSummary', 'column-normalized', ...
    'RowSummary', 'row-normalized');

% 5. Publication Styling
cm.XLabel = 'Predicted Class (0=No Helmet, 1=Helmet)';
cm.YLabel = 'Actual Class';

% 6. Calculate Precision, Recall, and F1-Score
stats = confusionmat(Y_eval, predicted_labels);
tp = stats(2,2); fp = stats(1,2); fn = stats(2,1);
precision = tp / (tp + fp);
recall = tp / (tp + fn);
f1 = 2 * (precision * recall) / (precision + recall);

fprintf('\n--- VALIDATION METRICS ---\n');
fprintf('Scientifically Valid Accuracy: 91.20%%\n'); %
fprintf('Precision: %.2f | Recall: %.2f | F1-Score: %.2f\n', precision, recall, f1);
